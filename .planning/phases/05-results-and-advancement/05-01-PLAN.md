       ```typescript
         score_b: z.number().int().min(0),
       })

         walkover: z.enum(['a', 'b']),
       })
       export const MatchResultPayloadSchema = z.union([ScorePayloadSchema, WalkoverPayloadSchema])
       ```


    3. `validateScore(match, result)` — per D-06:
       - If walkover payload, return immediately (always valid)
       - For score payload:
         - If `sets_to_win === 1`: validate as legs-only — winner must have exactly `Math.ceil(legs_per_set / 2)` legs, loser must have fewer
         - If `sets_to_win > 1`: validate as sets — winner must have exactly `sets_to_win` sets, loser must have fewer
       - Throw `Error("ScoreInvalid: ...")` with descriptive message on failure

    4. `StandingEntrySchema` — Zod schema:
       ```typescript
       export const StandingEntrySchema = z.object({
         team_id: z.string(),
         played: z.number().int(),
         wins: z.number().int(),
         losses: z.number().int(),
         points: z.number().int(),
         legs_won: z.number().int(),
         legs_lost: z.number().int(),
         leg_diff: z.number().int(),
       })
       export type StandingEntry = z.infer<typeof StandingEntrySchema>
       ```

    5. `computeStandings(matches)` — per D-07/D-08:
       - Filter matches to `status IN ('done', 'walkover')` only (exclude pending, bye)
       - For each match: winner gets SCORING_RULES.WIN (or WALKOVER_WIN), loser gets SCORING_RULES.LOSS (or WALKOVER_LOSS)
       - Accumulate legs_won/legs_lost from score_a/score_b (walkover = 0-0)
       - Return sorted by `breakTie` comparator
       - Input type: `Array<{ team_a_id: string | null; team_b_id: string | null; score_a: number | null; score_b: number | null; status: string }>`

    6. `breakTie(a, b, matches)` — per D-09:
       - Compare by points DESC
       - Then by leg_diff DESC
       - Then by head-to-head result (find direct match between a.team_id and b.team_id in provided matches; winner ranks higher)
