    Define `internalMatchRepo` with these functions (all take `sql: Sql` as first param):

    1. **`lockMatchForUpdate(sql, matchId)`** — SELECT ... FOR UPDATE on match row. Returns match data including `id, phase_id, team_a_id, team_b_id, status, sets_to_win, legs_per_set, score_a, score_b, bracket_info_id, round_robin_info_id, event_match_id`. Zod-validate the result. Throw `Error("NotFound")` if no row. Throw `Error("MatchAlreadyDone")` if `status !== 'pending'`.

    2. **`updateMatchResult(sql, matchId, scoreA, scoreB, status)`** — `UPDATE match SET score_a = $scoreA, score_b = $scoreB, status = $status WHERE id = $matchId`. Status is `'done'` or `'walkover'`.

    3. **`advanceWinnerInBracket(sql, matchId, winnerTeamId)`** — Read `bracket_match_info` for this match's `bracket_info_id` to get `winner_goes_to_info_id` and `winner_goes_to_slot`. If not null, update the target match: `UPDATE match SET team_a_id = $winnerTeamId WHERE bracket_info_id = $targetInfoId` (if slot='a') or `SET team_b_id` (if slot='b'). SQL:
    ```sql
    WITH src AS (
      SELECT bi.winner_goes_to_info_id, bi.winner_goes_to_slot
      FROM match m
      JOIN bracket_match_info bi ON bi.id = m.bracket_info_id
      WHERE m.id = $matchId
    )
    UPDATE match
    SET team_a_id = CASE WHEN src.winner_goes_to_slot = 'a' THEN $winnerTeamId ELSE match.team_a_id END,
        team_b_id = CASE WHEN src.winner_goes_to_slot = 'b' THEN $winnerTeamId ELSE match.team_b_id END
    FROM src
    WHERE match.bracket_info_id = src.winner_goes_to_info_id
      AND src.winner_goes_to_info_id IS NOT NULL
    ```
    Return the number of rows updated (0 if winner_goes_to is null = final match winner / champion).

    4. **`advanceLoserInBracket(sql, matchId, loserTeamId)`** — Same pattern but using `loser_goes_to_info_id` and `loser_goes_to_slot`. Only relevant for double-elimination WB matches and some LB matches.
    ```sql
    WITH src AS (
      SELECT bi.loser_goes_to_info_id, bi.loser_goes_to_slot
      FROM match m
      JOIN bracket_match_info bi ON bi.id = m.bracket_info_id
      WHERE m.id = $matchId
    )
    UPDATE match
    SET team_a_id = CASE WHEN src.loser_goes_to_slot = 'a' THEN $loserTeamId ELSE match.team_a_id END,
        team_b_id = CASE WHEN src.loser_goes_to_slot = 'b' THEN $loserTeamId ELSE match.team_b_id END
    FROM src
    WHERE match.bracket_info_id = src.loser_goes_to_info_id
      AND src.loser_goes_to_info_id IS NOT NULL
    ```

    5. **`checkPhaseComplete(sql, phaseId)`** — Returns `{ total: number, finished: number }`:
    ```sql
    SELECT COUNT(*)::int AS total,
           COUNT(CASE WHEN status IN ('done', 'walkover', 'bye') THEN 1 END)::int AS finished
    FROM match WHERE phase_id = $phaseId
    ```
    Zod-validate with `z.object({ total: z.number(), finished: z.number() })`.

    6. **`getPhaseQualifiers(sql, phaseId, qualifiersCount)`** — For round-robin phases, get the top N teams per group after all matches complete. Returns `string[]` (team_ids in rank order). This uses `computeStandings` from domain:
    ```sql
    SELECT m.team_a_id, m.team_b_id, m.score_a, m.score_b, m.status,
           rri.group_number
    FROM match m
    JOIN round_robin_match_info rri ON rri.id = m.round_robin_info_id
    WHERE m.phase_id = $phaseId
    ```
    Group by `group_number`, compute standings per group, take top N from each group. Return team_ids in interleaved order: group0-rank1, group1-rank1, group0-rank2, group1-rank2, etc.

    7. **`seedNextPhase(sql, phaseId, teamIds)`** — Find the next phase (by position) in the same tournament, then assign team_ids to the first round matches of that phase. For bracket phases: fill `team_a_id`/`team_b_id` by matching `seed_a`/`seed_b` in `bracket_match_info`. For round-robin phases: fill by `slot_a`/`slot_b` in `round_robin_match_info`.
    ```sql
    -- Find next phase
    SELECT p2.id FROM phase p1 JOIN phase p2
      ON p2.tournament_id = p1.tournament_id AND p2.position = p1.position + 1
    WHERE p1.id = $phaseId
    ```

    8. **`getMatchBracketInfo(sql, matchId)`** — Returns bracket and bracket_info_id for a match. Used to check if it's a GF match.
    ```sql
    SELECT bi.bracket, bi.id AS info_id, bi.winner_goes_to_info_id
    FROM match m JOIN bracket_match_info bi ON bi.id = m.bracket_info_id
    WHERE m.id = $matchId
    ```

    9. **`createResetMatch(sql, matchId, eventId)`** — Creates a bracket reset match dynamically (per D-12). Inside advisory lock on event:
    ```sql
    SELECT pg_advisory_xact_lock(hashtext($eventId))

    Then read MAX(event_match_id) for the event, create new bracket_match_info (bracket='GF', winner_goes_to=null), create new match row with event_match_id = max+1, and wire the original GF match's winner_goes_to to the new reset match.

    10. **`lookupMatchByEventMatchId(sql, eventId, eventMatchId)`** — For the hub tile lookup (per D-01/D-03):

            FROM team_member tm JOIN player pl ON pl.id = tm.player_id
            WHERE tm.team_id = m.team_a_id) AS team_a_name,
           (SELECT string_agg(pl.first_name || ' ' || pl.last_name, ' / ' ORDER BY pl.last_name)
            FROM team_member tm JOIN player pl ON pl.id = tm.player_id

            FROM team_member tm JOIN player pl ON pl.id = tm.player_id
            WHERE tm.team_id = m.referee_team_id) AS referee_name
    FROM match m
    JOIN phase p ON p.id = m.phase_id
    JOIN tournament t ON t.id = p.tournament_id
    WHERE m.event_match_id = $eventMatchId AND t.event_id = $eventId
    ```

    ```typescript
    import {
      getMatchRepositoryWithSql,
      getLaunchRepositoryWithSql,
      sql,
    } from "@darts-management/db"
    import {
      validateScore,
      computeStandings,
      type MatchResultPayload,
    } from "@darts-management/domain"
    import { trace } from "@opentelemetry/api"

    type Sql = typeof sql

    export async function submitMatchResult(
      matchId: string,
      payload: MatchResultPayload,
      userRoles: Array<{ entityId: string; role: string }>,
    ): Promise<void> {
      trace.getActiveSpan()?.setAttribute("match.id", matchId)

      await sql.begin(async (rawTx) => {
        const tx = rawTx as unknown as Sql
        const matchRepo = getMatchRepositoryWithSql(tx)
        const launchRepo = getLaunchRepositoryWithSql(tx)

        // 1. Lock match row (prevents double-submission per Pitfall 1)
        const match = await matchRepo.lockMatchForUpdate(matchId)

        // 2. Authz: get entity_id from tournament chain
        // match.phase_id → phase.tournament_id → tournament → event.entity_id
        // We need the entity_id. Load from launch repo which already has this pattern.
        const tournamentInfo = await tx`
          SELECT t.id, e.entity_id
          FROM phase p
          JOIN tournament t ON t.id = p.tournament_id
          JOIN event e ON e.id = t.event_id
          WHERE p.id = ${match.phase_id}
        `
        if (tournamentInfo.length === 0) throw new Error("NotFound")
        const entityId = tournamentInfo[0].entity_id as string

        const hasAccess = userRoles.some(
          (r) =>
            r.entityId === entityId &&
            ["adminTournoi", "adminClub", "adminComite", "adminLigue", "adminFederal"].includes(r.role),
        )
        if (!hasAccess) throw new Error("Forbidden")

        // 3. Validate score (throws ScoreInvalid: if bad)
        validateScore(
          { sets_to_win: match.sets_to_win, legs_per_set: match.legs_per_set },
          payload,
        )

        // 4. Determine winner/loser and update match
        let scoreA: number
        let scoreB: number
        let status: "done" | "walkover"
        let winnerTeamId: string
        let loserTeamId: string

        if ("walkover" in payload) {
          scoreA = 0
          scoreB = 0
          status = "walkover"
          winnerTeamId = payload.walkover === "a" ? match.team_b_id! : match.team_a_id!
          loserTeamId = payload.walkover === "a" ? match.team_a_id! : match.team_b_id!
        } else {
          scoreA = payload.score_a
          scoreB = payload.score_b
          status = "done"
          winnerTeamId = scoreA > scoreB ? match.team_a_id! : match.team_b_id!
          loserTeamId = scoreA > scoreB ? match.team_b_id! : match.team_a_id!
        }

        await matchRepo.updateMatchResult(matchId, scoreA, scoreB, status)

        // 5. Bracket advancement (if bracket match)
        if (match.bracket_info_id) {
          await matchRepo.advanceWinnerInBracket(matchId, winnerTeamId)
          await matchRepo.advanceLoserInBracket(matchId, loserTeamId)

          // 6. Check for GF bracket reset (D-12)
          const bracketInfo = await matchRepo.getMatchBracketInfo(matchId)
          if (bracketInfo && bracketInfo.bracket === "GF" && bracketInfo.winner_goes_to_info_id === null) {
            // This is the GF. If winner came from Lower bracket, create reset match.
            // The Lower bracket winner is in slot B of GF (convention from double-bracket-elimination.ts)
            // If winnerTeamId === match.team_b_id (Lower bracket winner), create reset
            if (winnerTeamId === match.team_b_id) {
              // Need event_id for advisory lock on event_match_id
              const eventId = tournamentInfo[0].entity_id as string // Wrong — need event_id
              const eventRow = await tx`SELECT e.id FROM tournament t JOIN event e ON e.id = t.event_id WHERE t.id = ${tournamentInfo[0].id}`
              if (eventRow.length > 0) {
                await matchRepo.createResetMatch(matchId, eventRow[0].id as string)
              }
            }
          }
        }

        // 7. Check if phase is complete → trigger next phase
        const { total, finished } = await matchRepo.checkPhaseComplete(match.phase_id)
        if (total === finished) {
          // Phase complete — seed qualifiers into next phase
          // For round-robin: compute standings, take top qualifiers
          // For bracket: winner of final goes to next phase
          if (match.round_robin_info_id) {
            // Get phase config to know how many qualifiers per group
            const phaseConfig = await tx`
              SELECT qualifiers_per_group FROM phase WHERE id = ${match.phase_id}
            `
            const qualifiersPerGroup = (phaseConfig[0]?.qualifiers_per_group as number) ?? 1
            const qualifiedTeams = await matchRepo.getPhaseQualifiers(match.phase_id, qualifiersPerGroup)
            if (qualifiedTeams.length > 0) {
              await matchRepo.seedNextPhase(match.phase_id, qualifiedTeams)
            }
          } else if (match.bracket_info_id) {
            // For bracket phases: the winner of the final match advances
            // Only if there's a next phase
            await matchRepo.seedNextPhase(match.phase_id, [winnerTeamId])
          }
        }
      })
    }
    ```
