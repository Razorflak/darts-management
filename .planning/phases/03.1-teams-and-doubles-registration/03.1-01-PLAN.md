Créer la migration SQL complète. Ordre impératif des opérations :

1. ALTER TABLE player ADD COLUMN department VARCHAR(50);

2. CREATE TABLE team (
     id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
     created_at TIMESTAMPTZ NOT NULL DEFAULT now()
   );

3. CREATE TABLE team_member (
     team_id   UUID NOT NULL REFERENCES team(id) ON DELETE CASCADE,
     player_id UUID NOT NULL REFERENCES player(id) ON DELETE RESTRICT,
     PRIMARY KEY (team_id, player_id)
   );
   CREATE INDEX team_member_player_idx ON team_member(player_id);

4. ALTER TABLE tournament_registration ADD COLUMN team_id UUID REFERENCES team(id) ON DELETE RESTRICT;

5. Migration données existantes via CTE pur SQL (NE PAS utiliser de boucle applicative) :
   WITH reg_with_teams AS (
     SELECT id AS reg_id, player_id, gen_random_uuid() AS new_team_id
     FROM tournament_registration
   ),
   _ins_teams AS (
     INSERT INTO team (id) SELECT new_team_id FROM reg_with_teams
   ),
   _ins_members AS (
     INSERT INTO team_member (team_id, player_id)
     SELECT new_team_id, player_id FROM reg_with_teams
   )
   UPDATE tournament_registration r
   SET team_id = rt.new_team_id
   FROM reg_with_teams rt
   WHERE r.id = rt.reg_id;

6. ALTER TABLE tournament_registration ALTER COLUMN team_id SET NOT NULL;

7. ALTER TABLE tournament_registration DROP COLUMN player_id;

8. ALTER TABLE tournament_registration ADD CONSTRAINT reg_team_tournament_unique UNIQUE (tournament_id, team_id);

9. DROP INDEX IF EXISTS reg_player_idx;
   CREATE INDEX reg_team_idx ON tournament_registration(team_id);

ATTENTION : Les CTEs doivent insérer dans team ET team_member ET mettre à jour tournament_registration en une seule transaction atomique. Si une inscription existante a player_id NULL, la sauter (WHERE player_id IS NOT NULL dans le CTE).

const RosterMemberSchema = z.object({
  player_id: z.uuid(),
  first_name: z.string(),
  last_name: z.string(),
  department: z.string().nullable()
})

export const RosterEntrySchema = z.object({
  registration_id: z.uuid(),
  team_id: z.uuid(),
  members: z.preprocess(
    (val) => (typeof val === 'string' ? JSON.parse(val) : val),
    z.array(RosterMemberSchema)

  checked_in: z.boolean(),
  registered_at: z.coerce.date()
})
export type RosterEntry = z.infer<typeof RosterEntrySchema>
